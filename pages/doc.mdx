```js
model Task {
    fields {
        description Text
    }

    actions {
        create createTask() with (description)
        get getTask(id)
        update updateTask(id) with (description)
        delete deleteTask(id)
        list listTask() {
            @orderBy(createdAt: desc)
        }
    }

    @permission(
        actions: [create, get, update, delete, list],
        expression: true
    )
}
```
